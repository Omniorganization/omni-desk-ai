name: Release Build

on:
  workflow_dispatch:
    inputs:
      release_channel:
        description: "candidate permits external-evidence audit; real-ga requires fail-closed evidence"
        type: choice
        required: true
        default: candidate
        options:
          - candidate
          - real-ga
      external_evidence_run_id:
        description: "Enforced Main Verification run id that uploaded the complete bound external evidence artifact"
        required: false
        default: ""
      external_evidence_artifact_name:
        description: "Bound external evidence artifact name from Main Verification"
        required: false
        default: external-ga-evidence-bound

permissions:
  contents: write
  actions: read
  packages: write
  id-token: write
  attestations: write

env:
  EXPECTED_VERSION: 1.12.7+root-monorepo-production-ga-candidate
  RELEASE_CHANNEL: ${{ inputs.release_channel || 'candidate' }}

jobs:
  web-admin-release:
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: apps/web-admin-next
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065
        with:
          python-version: "3.11"
      - uses: actions/setup-node@49933ea5288caeca8642d1e84afbd3f7d6820020
        with:
          node-version: "22"
          cache: npm
          cache-dependency-path: apps/web-admin-next/package-lock.json
      - run: npm ci
      - run: npm run typecheck
      - run: npm test
      - run: npm run build
      - name: Build and export Web Admin OCI image
        working-directory: .
        run: |
          set -euo pipefail
          mkdir -p "$RUNNER_TEMP/release-web-admin/payload"
          docker pull node:22-bookworm-slim
          NODE_BASE_IMAGE="$(docker image inspect --format '{{index .RepoDigests 0}}' node:22-bookworm-slim)"
          test -n "$NODE_BASE_IMAGE"
          docker build --build-arg "NODE_BASE_IMAGE=$NODE_BASE_IMAGE" -f apps/web-admin-next/Dockerfile -t omnidesk-web-admin:${{ github.sha }} apps/web-admin-next
          docker save omnidesk-web-admin:${{ github.sha }} | gzip -9 > "$RUNNER_TEMP/release-web-admin/payload/omnidesk-web-admin-${{ github.sha }}.tar.gz"
          python scripts/write_native_artifact_manifest.py \
            --platform web-admin \
            --artifact-root "$RUNNER_TEMP/release-web-admin/payload" \
            --source-commit "$GITHUB_SHA" \
            --output "$RUNNER_TEMP/release-web-admin/native-artifact-manifest.json"
      - uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02
        with:
          name: release-web-admin
          path: ${{ runner.temp }}/release-web-admin
          if-no-files-found: error

  desktop-release:
    strategy:
      fail-fast: false
      matrix:
        include:
          - os: ubuntu-latest
            artifact: desktop-linux
          - os: macos-latest
            artifact: desktop-macos
          - os: windows-latest
            artifact: desktop-windows
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065
        with:
          python-version: "3.11"
      - uses: actions/setup-node@49933ea5288caeca8642d1e84afbd3f7d6820020
        with:
          node-version: "22"
          cache: npm
          cache-dependency-path: apps/desktop-tauri/package-lock.json
      - uses: dtolnay/rust-toolchain@56f84321dbccf38fb67ce29ab63e4754056677e0
        with:
          toolchain: stable
          components: clippy
      - name: Install Linux Tauri dependencies
        if: runner.os == 'Linux'
        run: sudo apt-get update && sudo apt-get install -y libgtk-3-dev libwebkit2gtk-4.1-dev libayatana-appindicator3-dev librsvg2-dev patchelf
      - run: npm ci
        working-directory: apps/desktop-tauri
      - run: npm run typecheck
        working-directory: apps/desktop-tauri
      - run: npm test
        working-directory: apps/desktop-tauri
      - run: cargo test --locked --manifest-path apps/desktop-tauri/src-tauri/Cargo.toml
      - run: cargo clippy --locked --manifest-path apps/desktop-tauri/src-tauri/Cargo.toml -- -D warnings
      - run: npm run tauri:build
        working-directory: apps/desktop-tauri
      - name: Collect digest-bound Tauri bundles
        shell: bash
        run: |
          set -euo pipefail
          bundle_root="apps/desktop-tauri/src-tauri/target/release/bundle"
          test -d "$bundle_root"
          test -n "$(find "$bundle_root" -type f -print -quit)"
          mkdir -p "$RUNNER_TEMP/release-${{ matrix.artifact }}/payload"
          (cd "$bundle_root" && tar -cf - .) | (cd "$RUNNER_TEMP/release-${{ matrix.artifact }}/payload" && tar -xf -)
          python scripts/write_native_artifact_manifest.py \
            --platform "${{ matrix.artifact }}" \
            --artifact-root "$RUNNER_TEMP/release-${{ matrix.artifact }}/payload" \
            --source-commit "$GITHUB_SHA" \
            --output "$RUNNER_TEMP/release-${{ matrix.artifact }}/native-artifact-manifest.json"
      - uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02
        with:
          name: release-${{ matrix.artifact }}
          path: ${{ runner.temp }}/release-${{ matrix.artifact }}
          if-no-files-found: error

  mobile-android-release:
    runs-on: ubuntu-latest
    timeout-minutes: 35
    defaults:
      run:
        working-directory: apps/mobile-flutter
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065
        with:
          python-version: "3.11"
      - uses: subosito/flutter-action@e938fdf56512cc96ef2f93601a5a40bde3801046
        with:
          flutter-version: "3.38.x"
          cache: true
      - run: flutter pub get
      - run: dart analyze
      - run: flutter test
      - name: Prepare Android release SDK packages
        run: |
          set -euo pipefail
          sdkmanager="${ANDROID_HOME}/cmdline-tools/latest/bin/sdkmanager"
          set +o pipefail
          yes | "$sdkmanager" --licenses >/dev/null
          license_status=$?
          set -o pipefail
          test "$license_status" -eq 0
          "$sdkmanager" "platforms;android-36" "build-tools;36.0.0" "ndk;28.2.13676358"
      - name: Restore Android release signing key
        env:
          OMNI_ANDROID_KEYSTORE_BASE64: ${{ secrets.OMNI_ANDROID_KEYSTORE_BASE64 }}
          OMNI_ANDROID_KEYSTORE_PASSWORD: ${{ secrets.OMNI_ANDROID_KEYSTORE_PASSWORD }}
          OMNI_ANDROID_KEY_ALIAS: ${{ secrets.OMNI_ANDROID_KEY_ALIAS }}
          OMNI_ANDROID_KEY_PASSWORD: ${{ secrets.OMNI_ANDROID_KEY_PASSWORD }}
        run: |
          set -euo pipefail
          test -n "$OMNI_ANDROID_KEYSTORE_BASE64"
          test -n "$OMNI_ANDROID_KEYSTORE_PASSWORD"
          test -n "$OMNI_ANDROID_KEY_ALIAS"
          test -n "$OMNI_ANDROID_KEY_PASSWORD"
          printf '%s' "$OMNI_ANDROID_KEYSTORE_BASE64" | base64 -d > "$RUNNER_TEMP/omnidesk-release.keystore"
          {
            echo "OMNI_ANDROID_KEYSTORE=$RUNNER_TEMP/omnidesk-release.keystore"
            echo "OMNI_ANDROID_KEYSTORE_PASSWORD=$OMNI_ANDROID_KEYSTORE_PASSWORD"
            echo "OMNI_ANDROID_KEY_ALIAS=$OMNI_ANDROID_KEY_ALIAS"
            echo "OMNI_ANDROID_KEY_PASSWORD=$OMNI_ANDROID_KEY_PASSWORD"
          } >> "$GITHUB_ENV"
      - name: Restore Android Firebase config
        env:
          OMNI_ANDROID_GOOGLE_SERVICES_JSON: ${{ secrets.OMNI_ANDROID_GOOGLE_SERVICES_JSON }}
        run: |
          set -euo pipefail
          test -n "$OMNI_ANDROID_GOOGLE_SERVICES_JSON"
          printf '%s' "$OMNI_ANDROID_GOOGLE_SERVICES_JSON" > android/app/google-services.json
      - run: flutter build appbundle --release
        timeout-minutes: 20
      - name: Verify and collect signed Android App Bundle
        working-directory: .
        run: |
          set -euo pipefail
          aab="$(find apps/mobile-flutter/build/app/outputs/bundle/release -type f -name '*.aab' -print -quit)"
          test -n "$aab"
          jarsigner -verify -verbose -certs "$aab"
          mkdir -p "$RUNNER_TEMP/release-mobile-android/payload"
          cp "$aab" "$RUNNER_TEMP/release-mobile-android/payload/"
          python scripts/write_native_artifact_manifest.py \
            --platform mobile-android \
            --artifact-root "$RUNNER_TEMP/release-mobile-android/payload" \
            --source-commit "$GITHUB_SHA" \
            --output "$RUNNER_TEMP/release-mobile-android/native-artifact-manifest.json"
      - uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02
        with:
          name: release-mobile-android
          path: ${{ runner.temp }}/release-mobile-android
          if-no-files-found: error

  mobile-ios-release:
    runs-on: macos-latest
    defaults:
      run:
        working-directory: apps/mobile-flutter
    steps:
      - uses: actions/checkout@34e114876b0b11c390a56381ad16ebd13914f8d5
      - uses: actions/setup-python@a26af69be951a213d495a4c3e4e4022e16d87065
        with:
          python-version: "3.11"
      - uses: subosito/flutter-action@e938fdf56512cc96ef2f93601a5a40bde3801046
        with:
          flutter-version: "3.38.x"
          cache: true
      - run: flutter pub get
      - run: cd ios && pod install
      - name: Restore iOS distribution signing material
        env:
          OMNI_IOS_DISTRIBUTION_CERT_BASE64: ${{ secrets.OMNI_IOS_DISTRIBUTION_CERT_BASE64 }}
          OMNI_IOS_DISTRIBUTION_CERT_PASSWORD: ${{ secrets.OMNI_IOS_DISTRIBUTION_CERT_PASSWORD }}
          OMNI_IOS_PROVISIONING_PROFILE_BASE64: ${{ secrets.OMNI_IOS_PROVISIONING_PROFILE_BASE64 }}
          OMNI_IOS_EXPORT_OPTIONS_PLIST_BASE64: ${{ secrets.OMNI_IOS_EXPORT_OPTIONS_PLIST_BASE64 }}
          OMNI_IOS_KEYCHAIN_PASSWORD: ${{ secrets.OMNI_IOS_KEYCHAIN_PASSWORD }}
        run: |
          set -euo pipefail
          if [[ "$RELEASE_CHANNEL" == "real-ga" ]]; then
            test -n "$OMNI_IOS_DISTRIBUTION_CERT_BASE64"
            test -n "$OMNI_IOS_DISTRIBUTION_CERT_PASSWORD"
            test -n "$OMNI_IOS_PROVISIONING_PROFILE_BASE64"
            test -n "$OMNI_IOS_EXPORT_OPTIONS_PLIST_BASE64"
            test -n "$OMNI_IOS_KEYCHAIN_PASSWORD"
          fi
          if [[ -n "${OMNI_IOS_DISTRIBUTION_CERT_BASE64:-}" ]]; then
            python - <<'PY'
          import base64
          import os
          from pathlib import Path
          for env_name, target in (
              ("OMNI_IOS_DISTRIBUTION_CERT_BASE64", Path(os.environ["RUNNER_TEMP"]) / "distribution.p12"),
              ("OMNI_IOS_PROVISIONING_PROFILE_BASE64", Path(os.environ["RUNNER_TEMP"]) / "omnidesk.mobileprovision"),
              ("OMNI_IOS_EXPORT_OPTIONS_PLIST_BASE64", Path(os.environ["RUNNER_TEMP"]) / "ExportOptions.plist"),
          ):
              target.write_bytes(base64.b64decode(os.environ[env_name]))
          PY
            keychain="$RUNNER_TEMP/omnidesk.keychain-db"
            security create-keychain -p "$OMNI_IOS_KEYCHAIN_PASSWORD" "$keychain"
            security set-keychain-settings -lut 21600 "$keychain"
            security unlock-keychain -p "$OMNI_IOS_KEYCHAIN_PASSWORD" "$keychain"
            security default-keychain -s "$keychain"
            security import "$RUNNER_TEMP/distribution.p12" -k "$keychain" -P "$OMNI_IOS_DISTRIBUTION_CERT_PASSWORD" -T /usr/bin/codesign
            security set-key-partition-list -S apple-tool:,apple: -s -k "$OMNI_IOS_KEYCHAIN_PASSWORD" "$keychain"
            mkdir -p "$HOME/Library/MobileDevice/Provisioning Profiles"
            cp "$RUNNER_TEMP/omnidesk.mobileprovision" "$HOME/Library/MobileDevice/Provisioning Profiles/omnidesk.mobileprovision"
            security find-identity -v -p codesigning "$keychain"
            echo "OMNI_IOS_SIGNED_BUILD=1" >> "$GITHUB_ENV"
          else
            echo "OMNI_IOS_SIGNED_BUILD=0" >> "$GITHUB_ENV"
          fi
      - name: Build signed IPA or candidate no-codesign app
        run: |
          set -euo pipefail
          if [[ "$OMNI_IOS_SIGNED_BUILD" == "1" ]]; then
            flutter build ipa --release --export-options-plist="$RUNNER_TEMP/ExportOptions.plist"
            ipa="$(find build/ios/ipa -type f -name '*.ipa' -print -quit)"
            test -n "$ipa"
            rm -rf "$RUNNER_TEMP/ipa-verify"
            mkdir -p "$RUNNER_TEMP/ipa-verify"
            unzip -q "$ipa" -d "$RUNNER_TEMP/ipa-verify"
            app="$(find "$RUNNER_TEMP/ipa-verify/Payload" -maxdepth 1 -type d -name '*.app' -print -quit)"
            test -n "$app"
            codesign --verify --deep --strict --verbose=2 "$app"
          else
            flutter build ios --release --no-codesign
          fi
      - name: Collect digest-bound iOS payload
        working-directory: .
        run: |
          set -euo pipefail
          mkdir -p "$RUNNER_TEMP/release-mobile-ios/payload"
          if [[ "$OMNI_IOS_SIGNED_BUILD" == "1" ]]; then
            ipa="$(find apps/mobile-flutter/build/ios/ipa -type f -name '*.ipa' -print -quit)"
            cp "$ipa" "$RUNNER_TEMP/release-mobile-ios/payload/"
          else
            ditto -c -k --sequesterRsrc --keepParent \
              apps/mobile-flutter/build/ios/iphoneos/Runner.app \
              "$RUNNER_TEMP/release-mobile-ios/payload/Runner-candidate-nosign.zip"
          fi
          python scripts/write_native_artifact_manifest.py \
            --platform mobile-ios \
            --artifact-root "$RUNNER_TEMP/release-mobile-ios/payload" \
            --source-commit "$GITHUB_SHA" \
            --output "$RUNNER_TEMP/release-mobile-ios/native-artifact-manifest.json"
      - name: Remove temporary iOS signing material
        if: always()
        run: |
          security delete-keychain "$RUNNER_TEMP/omnidesk.keychain-db" 2>/dev/null || true
          rm -f "$RUNNER_TEMP/distribution.p12" "$RUNNER_TEMP/omnidesk.mobileprovision" "$RUNNER_TEMP/ExportOptions.plist"
          rm -f "$HOME/Library/MobileDevice/Provisioning Profiles/omnidesk.mobileprovision"
      - uses: actions/upload-artifact@ea165f8d65b6e75b540449e92b4886f43607fa02
        with:
          name: release-mobile-ios
          path: ${{ runner.temp }}/release-mobile-ios
          if-no-files-found: error

      - name: Import iOS real-device evidence

      - name: Import tri-app live smoke evidence
        env:
          TRI_APP_ADMIN_TOKEN: ${{ secrets.TRI_APP_ADMIN_TOKEN }}

            --artifact-root "$RUNNER_TEMP/tri-app-native-input" \
